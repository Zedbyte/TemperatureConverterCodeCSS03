    DecimalFormat df = new DecimalFormat("##.####");
    DecimalFormat df2 = new DecimalFormat("##.##");
    
    double ctof(double c){
        double f;
        f = ((9*c)/5)+32;
        return f;
    }
    
    double ftoc(double f){
        double c;
        c=((f-32)*5)/9;
        return c;
    }
    
    void FormulaC() {
        
         String t1 = jTextField1.getText();
         String t2test = jTextField2.getText();
         
         double decimal = Double.parseDouble(t2test);
         String t2 =df2.format(decimal);
         
         String t3 = "(" + t1 + "°C × 9/5) + 32 = " + t2  +"°F";
         jLabel4.setText(t3);
    }
    
    void FormulaCEmpty() {
        String t3 = "(" + "" + "°C × 9/5) + 32 = " + ""  +"°F";
         jLabel4.setText(t3);
    }
    
     void FormulaF() {
         String t1 = jTextField1.getText();
         String t2test = jTextField2.getText();
         
         double decimal = Double.parseDouble(t2test);
         String t2 =df2.format(decimal);
         
         String t3 = "(" + t1 + "°F − 32) × 5/9 = " + t2 + "°C";
         jLabel4.setText(t3);
    }
     
     void FormulaFEmpty() {
        String t3 = "(" + "" + "°F − 32) × 5/9 = " + "" + "°C";

    }

    private void jComboBox1ActionPerformed(java.awt.event.ActionEvent evt) {                                           
        if(jComboBox1.getSelectedIndex()==0)
        {
            jComboBox2.setSelectedIndex(1);
            double f,c;
            c=Double.parseDouble(jTextField1.getText());
            f=ctof(c);
            jTextField2.setText(df.format(f));
            
            FormulaC();
        }
        else
        {
            jComboBox2.setSelectedIndex(0);
            
            double f,c;
            f=Double.parseDouble(jTextField1.getText());
            c=ftoc(f);
            jTextField2.setText(df.format(c));
            
            FormulaF();
        }

    }                                          

    private void jComboBox2ActionPerformed(java.awt.event.ActionEvent evt) {                                           
        if(jComboBox2.getSelectedIndex()==0) //value is Celsius
        {
            jComboBox1.setSelectedIndex(1);
            
            double f,c;

            c=ftoc(f);
            jTextField2.setText(df.format(c));

        }
        else
        {
            jComboBox1.setSelectedIndex(0); 
           
            double f,c;
            c=Double.parseDouble(jTextField1.getText()); //value is Fahrenheit
            f=ctof(c);
            jTextField2.setText(df.format(f));
            
        }
    }

    private void jTextField1KeyReleased(java.awt.event.KeyEvent evt) {                                        
        double c,f;
        String txt;
        txt=jTextField1.getText();
        if(txt.equals("")||txt.equals("-"))
        {    
            jTextField2.setText("");
            jTextField1.requestFocus();
            
             if(jComboBox1.getSelectedIndex()==0) {
                 FormulaCEmpty();
             }
             else {
                 FormulaFEmpty();
             }
        }
        else
        {
            if(jComboBox1.getSelectedIndex()==0) {
                c=Double.parseDouble(jTextField1.getText());
                f=ctof(c);
                jTextField2.setText(df.format(f));
                
                FormulaC();
            }
            else
            {
                f=Double.parseDouble(jTextField1.getText());
                c=ftoc(f);
                jTextField2.setText(df.format(c));
                
                FormulaF();
            }
        }
    }
